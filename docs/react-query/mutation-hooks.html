Create Custom Mutation Hooks
Published: Jul 24, 2022 | Last Modified: Jul 24, 2022
Here we will create 4 different custom hooks using the useMutation hook to handle POST, PUT, PATCH and DELETE requests.
Create some interfaces in interfaces/mutationConfig.ts which will be used to create our mutation hooks.
1
export interface IPostMutationConfig {
2
  path: string;
3
  enableAbort?: boolean;
4
  isPrivate?: boolean;
5
}
6


7
export interface IPutPatchDelMutationConfig {
8
  pathFn(id: number): string;
9
  enableAbort?: boolean;
10
  isPrivate?: boolean;
11
}
Create Post Mutation
To handle POST request, create hooks/usePostMutation.ts
1
import { AxiosError, AxiosRequestConfig } from "axios";
2
import { useMutation, UseMutationOptions } from "react-query";
3
import { IPostMutationConfig } from "./interfaces/mutationConfig";
4
import { usePrivateAxios, usePublicAxios } from "../../axios";
5


6
export const usePostMutation = <TResponse, TVariables>(
7
  config: IPostMutationConfig,
8
  options?: Omit<
9
    UseMutationOptions<TResponse, AxiosError, TVariables>,
10
    "mutationFn"
11
  >,
12
  axiosRequestConfig?: AxiosRequestConfig
13
) => {
14
  const { path, enableAbort = true, isPrivate = true } = config;
15
  const { publicPost } = usePublicAxios();
16
  const { privatePost } = usePrivateAxios();
17


18
  const mutationFn = (variables: TVariables) => {
19
    if (isPrivate) {
20
      return privatePost<TResponse, TVariables>(
21
        path,
22
        variables,
23
        { activateAbort: enableAbort },
24
        axiosRequestConfig
25
      );
26
    }
27
    return publicPost<TResponse, TVariables>(
28
      path,
29
      variables,
30
      { activateAbort: enableAbort },
31
      axiosRequestConfig
32
    );
33
  };
34


35
  return useMutation<TResponse, AxiosError, TVariables>(mutationFn, options);
36
};
